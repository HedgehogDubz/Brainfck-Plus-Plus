    return 0;
}  
void mov_right(){
    p++;
    if(p >= a.size()){
        a.push_back(0);
    }
}
bool mov_left(){
    p--;
    if(p < 0){
        std::cerr << "\033[31m" << "ERROR pointer below 0" << "\033[0m" << std::endl;
        return true;
    }
    return false;
}
